def get_queue_position(uid, qtype='code'):
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    if qtype == 'code':
        c.execute('SELECT COUNT(*) FROM code_queue WHERE status="waiting" AND id<=(SELECT COALESCE(MIN(id),0) FROM code_queue WHERE user_id=? AND status="waiting")', (uid,))
    else:
        c.execute('SELECT COUNT(*) FROM qr_queue WHERE status="waiting" AND id<=(SELECT COALESCE(MIN(id),0) FROM qr_queue WHERE user_id=? AND status="waiting")', (uid,))
    p = c.fetchone()[0]
    conn.close()
    return p

def add_to_queue(uid, qtype='code', phone=None):
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    if qtype == 'code':
        c.execute('SELECT id FROM code_queue WHERE user_id=? AND status="waiting"', (uid,))
        if c.fetchone():
            conn.close()
            return False, "Вы уже в очереди"
        c.execute('INSERT INTO code_queue (user_id,phone_number,status) VALUES (?,?,"waiting")', (uid, phone))
    else:
        c.execute('SELECT id FROM qr_queue WHERE user_id=? AND status="waiting"', (uid,))
        if c.fetchone():
            conn.close()
            return False, "Вы уже в очереди QR"
        c.execute('INSERT INTO qr_queue (user_id,status) VALUES (?,"waiting")', (uid,))
    conn.commit()
    conn.close()
    return True, "Добавлены в очередь"

def handle_code_success(call):

    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    c.execute('SELECT user_id, phone_number FROM code_queue WHERE id=?', (oid,))
    order = c.fetchone()
    if order:
        price = float(get_setting('code_price','4.00'))
        ht = int(get_setting('hold_time','300'))
        c.execute('UPDATE code_queue SET status="completed", completed_at=CURRENT_TIMESTAMP WHERE id=?', (oid,))
        c.execute('UPDATE users SET total_orders=total_orders+1, hold_balance=hold_balance+? WHERE user_id=?', (price, order[0]))
        conn.commit()
        conn.close()
        try: bot.send_message(order[0],
            f"🎉 **НОМЕР УСПЕШНО ВСТАЛ!**\n\n📱 `{order[1]}`\n✅ Статус: Подтверждено\n\n💰 Сумма в холде: **${price:.2f}**\n⏱ Холд: {ht//60} мин\n\n💎 Средства поступят через {ht//60} минут",
            parse_mode='Markdown')
        except: pass
        timer = threading.Timer(ht, process_hold_payment, args=[order[0], oid, price, 'code'])
        timer.start()
        if not hasattr(bot, 'hold_timers'): bot.hold_timers = {}
        bot.hold_timers[f'code_{oid}'] = {'timer': timer, 'user_id': order[0], 'amount': price}
        markup = types.InlineKeyboardMarkup()
        markup.add(types.InlineKeyboardButton("❌ Слетел", callback_data=f'code_failed_{oid}'))
        notification_text = f"✅ **ПОДТВЕРЖДЕНО**\n\n📱 Номер `{order[1]}`\nуспешно встал"
        if thread_id: send_to_group(notification_text, markup=markup, thread_id=thread_id)
        else:
            try: bot.edit_message_text(notification_text, call.message.chat.id, call.message.message_id, parse_mode='Markdown', reply_markup=markup)
            except: send_to_group(notification_text, markup=markup)
        bot.answer_callback_query(call.id, "✅ Подтверждено")
    else: bot.answer_callback_query(call.id, "❌ Заявка не найдена", show_alert=True)

def handle_qr_success(call):

    thread_id = call.message.message_thread_id if hasattr(call.message, 'message_thread_id') else None
    logger.info(f"✅ Холодник нажал 'Встал' для QR заявки #{qr_id}")

    c.execute('SELECT user_id, status, completed FROM qr_queue WHERE id=?', (qr_id,))
    qr = c.fetchone()
    if not qr:
        logger.error(f"❌ QR заявка #{qr_id} не найдена")
        bot.answer_callback_query(call.id, "❌ Заявка не найдена", show_alert=True)
        conn.close()
        return
    logger.info(f"🔍 ДО обновления QR #{qr_id}: status={qr[1]}, completed={qr[2]}")
    price = float(get_setting('qr_price','4.00'))
    ht = int(get_setting('hold_time','300'))
    c.execute("UPDATE qr_queue SET status='completed', completed=1, completed_at=CURRENT_TIMESTAMP WHERE id=?", (qr_id,))

    conn.commit()
    c.execute('SELECT status, completed FROM qr_queue WHERE id=?', (qr_id,))
    final_check = c.fetchone()
    c.execute('SELECT balance, hold_balance FROM users WHERE user_id=?', (qr[0],))
    user_check = c.fetchone()
    logger.info(f"🔍 ФИНАЛЬНАЯ проверка QR #{qr_id}: status={final_check[0]}, completed={final_check[1]}")
    logger.info(f"🔍 User {qr[0]}: balance={user_check[0]}, hold={user_check[1]}")
    conn.close()
    try:
        bot.send_message(qr[0],
            f"🎉 **QR УСПЕШНО ПРИНЯТ!**\n\n📷 QR-код\n✅ Вход подтвержден!\n\n"
            f"💰 Сумма в холде: **${price:.2f}**\n⏱ Холд: {ht//60} мин\n\n"
            f"💎 Средства поступят через {ht//60} минут",
            parse_mode='Markdown')
    except: pass
    timer = threading.Timer(ht, process_hold_payment, args=[qr[0], qr_id, price, 'qr'])
    timer.start()
    if not hasattr(bot, 'hold_timers'): bot.hold_timers = {}
    bot.hold_timers[f'qr_{qr_id}'] = {'timer': timer, 'user_id': qr[0], 'amount': price}
    markup = types.InlineKeyboardMarkup()
    markup.add(types.InlineKeyboardButton("❌ Слетел", callback_data=f'qr_failed_{qr_id}'))
    notification_text = f"✅ **ПОДТВЕРЖДЕНО**\n\n📷 QR-код успешно принят"
    if thread_id: send_to_group(notification_text, markup=markup, thread_id=thread_id)
    else:
        try: bot.edit_message_text(notification_text, call.message.chat.id, call.message.message_id, parse_mode='Markdown', reply_markup=markup)
        except: send_to_group(notification_text, markup=markup)

def process_hold_payment(user_id, order_id, amount, order_type='code'):
    try:
        logger.info(f"🕐 Холд завершен! Начисление ${amount} пользователю {user_id} за номер #{order_id} ({order_type})")
        conn = sqlite3.connect('max_bot.db')
        c = conn.cursor()
        if order_type == 'code': c.execute('SELECT status, phone_number FROM code_queue WHERE id=?', (order_id,))
        else: c.execute('SELECT status, completed FROM qr_queue WHERE id=?', (order_id,))
        order = c.fetchone()
        if not order: logger.error(f"❌ Номер #{order_id} не найден"); conn.close(); return
        status = order[0]
        is_completed = False
        if order_type == 'code': is_completed = (status == 'completed')
        else:
            if len(order) > 1: is_completed = (status == 'completed' and order[1] == 1)
            else: is_completed = (status == 'completed')
        logger.info(f"📊 Номер #{order_id}: status={status}, is_completed={is_completed}")
        if is_completed:
            c.execute('UPDATE users SET balance=balance+?, hold_balance=hold_balance-?, total_earned=total_earned+? WHERE user_id=?', (amount, amount, amount, user_id))

@bot.message_handler(func=lambda m: user_states.get(m.from_user.id,{}).get('state')=='admin_editing_balance')
def process_admin_edit_balance(message):
    if not is_admin(message.from_user.id): return
    us = user_states[message.from_user.id]
    uid = us['user_id']
    del user_states[message.from_user.id]
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    c.execute('SELECT balance, username FROM users WHERE user_id=?', (uid,))
    user = c.fetchone()
    if not user: bot.reply_to(message, "❌ Не найден"); conn.close(); return
    old_balance = user[0]
    inp = message.text.strip()
    try:
        if inp.startswith('+'): new_balance = old_balance + float(inp[1:]); diff = float(inp[1:]); action = "пополнен"
        elif inp.startswith('-'): diff = float(inp[1:]); new_balance = max(0, old_balance - diff); action = "уменьшен"
        else: new_balance = float(inp); diff = new_balance - old_balance; action = "установлен"
    except: bot.reply_to(message, "❌ Неверный формат"); conn.close(); return
    c.execute('UPDATE users SET balance=? WHERE user_id=?', (new_balance, uid))
    if diff != 0: c.execute('INSERT INTO transactions (user_id, amount, type, description) VALUES (?,?,"admin_edit",?)', (uid, abs(diff), f'Админ {action} баланс на ${abs(diff):.2f}'))
    conn.commit()
    conn.close()
    try: bot.send_message(uid, f"💰 **Баланс изменен**\n\n📝 Старый: ${old_balance:.2f}\n💎 Новый: **${new_balance:.2f}**\n📊 Изменение: {'+' if diff > 0 else ''}${diff:.2f}", parse_mode='Markdown')
    except: pass
    bot.reply_to(message, f"✅ Баланс обновлен!\n👤 @{user[1] or 'Нет'}\n💰 ${old_balance:.2f} → **${new_balance:.2f}**", parse_mode='Markdown', reply_markup=types.InlineKeyboardMarkup().add(types.InlineKeyboardButton("🔍 Поиск", callback_data='admin_search'), types.InlineKeyboardButton("◀️ В админ-панель", callback_data='admin_back')))
