def is_vip(user_id):
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    c.execute('SELECT vip_until FROM users WHERE user_id=?', (user_id,))
    result = c.fetchone()
    conn.close()
    if result and result[0]:
        try:
            vip_until = datetime.strptime(result[0], '%Y-%m-%d %H:%M:%S')
            return vip_until > datetime.now()
        except:
            return False
    return False

def get_vip_days_left(user_id):
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    c.execute('SELECT vip_until FROM users WHERE user_id=?', (user_id,))
    result = c.fetchone()
    conn.close()
    if result and result[0]:
        try:
            vip_until = datetime.strptime(result[0], '%Y-%m-%d %H:%M:%S')
            now = datetime.now()
            if vip_until > now:
                delta = vip_until - now
                return delta.days + 1
        except:
            pass
    return 0

def get_top_daily(limit=10):
    today = get_msk_time().strftime('%Y-%m-%d')
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    c.execute('SELECT user_id, earned FROM daily_stats WHERE date=? ORDER BY earned DESC LIMIT ?', (today, limit))
    top = c.fetchall()
    conn.close()
    return top

def get_top_weekly(limit=10):
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    c.execute('SELECT user_id, SUM(earned) as total FROM weekly_stats GROUP BY user_id ORDER BY total DESC LIMIT ?', (limit,))
    top = c.fetchall()
    conn.close()
    return top

def hide_username(username):
    if not username: return "Пользователь"
    half = len(username) // 2
    if half == 0: return username + "***"
    return username[:half] + "***"

def get_queue_position(uid, qtype='code'):
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    if qtype == 'code':
        c.execute('SELECT COUNT(*) FROM code_queue WHERE status="waiting" AND id<=(SELECT COALESCE(MIN(id),0) FROM code_queue WHERE user_id=? AND status="waiting")', (uid,))
    else:
        c.execute('SELECT COUNT(*) FROM qr_queue WHERE status="waiting" AND id<=(SELECT COALESCE(MIN(id),0) FROM qr_queue WHERE user_id=? AND status="waiting")', (uid,))
    p = c.fetchone()[0]
    conn.close()
    return p

def add_to_queue(uid, qtype='code', phone=None):
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    if qtype == 'code':
        c.execute('SELECT id FROM code_queue WHERE user_id=? AND status="waiting"', (uid,))
        if c.fetchone():
            conn.close()
            return False, "Вы уже в очереди"
        c.execute('INSERT INTO code_queue (user_id,phone_number,status) VALUES (?,?,"waiting")', (uid, phone))
    else:
        c.execute('SELECT id FROM qr_queue WHERE user_id=? AND status="waiting"', (uid,))
        if c.fetchone():
            conn.close()
            return False, "Вы уже в очереди QR"
        c.execute('INSERT INTO qr_queue (user_id,status) VALUES (?,"waiting")', (uid,))
    conn.commit()
    conn.close()
    return True, "Добавлены в очередь"

@bot.message_handler(commands=['start'])
def start_command(message):
    uid = message.from_user.id
    args = message.text.split()
    if len(args) > 1 and args[1].startswith('ref_'):
        ref_code = args[1][4:]
        conn = sqlite3.connect('max_bot.db')
        c = conn.cursor()
        c.execute('SELECT user_id FROM users WHERE referral_code=? AND user_id!=?', (ref_code, uid))
        referrer = c.fetchone()
        if referrer:
            c.execute('SELECT id FROM referrals WHERE referred_id=?', (uid,))
            if not c.fetchone():
                c.execute('INSERT INTO referrals (referrer_id, referred_id) VALUES (?,?)', (referrer[0], uid))
                conn.commit()
                try:
                    bot.send_message(referrer[0],
                        "🎉 <b>Новый реферал!</b>\n\nПо вашей ссылке зарегистрировался новый пользователь!\n\n💰 Вы будете получать $0.10 за каждый успешно сданный им номер!",
                        parse_mode='HTML')
                except: pass
        conn.close()
    
    if not check_subscription(uid):
        markup = types.InlineKeyboardMarkup(row_width=1)
        if PRIVATE_CHANNEL:
            try:
                invite_link = bot.export_chat_invite_link(CHANNEL_ID)
                markup.add(types.InlineKeyboardButton("📢 Подписаться на канал", url=invite_link))
            except:

def show_profile(message):
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    c.execute('SELECT * FROM users WHERE user_id=?', (message.chat.id,))
    user = c.fetchone()
    
    if not user: 
        show_main_menu(message.chat.id, message.message_id)
        conn.close()
        return
    
    c.execute('''SELECT COUNT(*) as t, SUM(CASE WHEN status="completed" THEN 1 ELSE 0 END) as ok, SUM(CASE WHEN status="failed" THEN 1 ELSE 0 END) as fl, SUM(CASE WHEN status IN ("taken","waiting") THEN 1 ELSE 0 END) as ip FROM code_queue WHERE user_id=?''', (message.chat.id,))
    code_stats = c.fetchone()
    
    c.execute('''SELECT COUNT(*) as t, SUM(CASE WHEN completed=1 THEN 1 ELSE 0 END) as ok, SUM(CASE WHEN status="failed" THEN 1 ELSE 0 END) as fl, SUM(CASE WHEN status IN ("taken","waiting") THEN 1 ELSE 0 END) as ip FROM qr_queue WHERE user_id=?''', (message.chat.id,))
    qr_stats = c.fetchone()
    
    total_codes = code_stats[0] if code_stats and code_stats[0] else 0
    ok_codes = code_stats[1] if code_stats and code_stats[1] else 0
    fl_codes = code_stats[2] if code_stats and code_stats[2] else 0
    ip_codes = code_stats[3] if code_stats and code_stats[3] else 0
    
    total_qr = qr_stats[0] if qr_stats and qr_stats[0] else 0
    ok_qr = qr_stats[1] if qr_stats and qr_stats[1] else 0
    fl_qr = qr_stats[2] if qr_stats and qr_stats[2] else 0
    ip_qr = qr_stats[3] if qr_stats and qr_stats[3] else 0
    
    total = total_codes + total_qr
    ok = ok_codes + ok_qr
    fl = fl_codes + fl_qr
    ip = ip_codes + ip_qr
    rate = int(ok/total*100) if total > 0 else 0
    
    c.execute('SELECT phone_number, status FROM code_queue WHERE user_id=? AND status IN ("taken","waiting") ORDER BY created_at DESC', (message.chat.id,))
    active_codes = c.fetchall() or []
    
    c.execute('SELECT id, status FROM qr_queue WHERE user_id=? AND status IN ("taken","waiting") ORDER BY created_at DESC', (message.chat.id,))
    active_qrs = c.fetchall() or []
    
    c.execute('SELECT amount, description FROM transactions WHERE user_id=? ORDER BY created_at DESC LIMIT 1', (message.chat.id,))
    lt = c.fetchone()
    
    c.execute('SELECT COUNT(*) FROM withdraw_requests WHERE user_id=? AND status="pending"', (message.chat.id,))
    pw_result = c.fetchone()
    pw = pw_result[0] if pw_result else 0
    
    conn.close()
    
    status_text = 'Активен' if ip > 0 else 'Ожидает'
    
    vip_status = ""
    if is_vip(message.chat.id):
        days = get_vip_days_left(message.chat.id)
        vip_status = f"👑 VIP: ✅ Активен ({days} дн.)\n"
    else:
        vip_status = "👑 VIP: ❌ Неактивен\n"
    
    text = (
        f"👤 <b>МОЙ ПРОФИЛЬ</b>\n\n"
        f"{vip_status}"
        f"<b>Информация</b>\n"
        f"├─ ID: <code>{user[0]}</code>\n"
        f"├─ @{user[1] or 'Не указан'}\n"
        f"└─ Статус: {status_text}\n\n"
        f"<b>Финансы</b>\n"
        f"├─ Баланс: <b>${user[2]:.2f}</b>\n"
        f"├─ В холде: <b>${user[4]:.2f}</b>\n"
        f"├─ Заработано: ${user[3]:.2f}\n"
        f"└─ Выведено: ${user[5]:.2f}\n\n"
        f"<b>Статистика</b>\n"
        f"├─ Всего заказов: {total}\n"
        f"├─ Успешно: {ok} ({rate}%)\n"
        f"├─ В работе: {ip}\n"
        f"└─ Отменено: {fl}"
    )
    
    if active_codes:
        text += "\n\n<b>Номера в работе:</b>\n"
        for num in active_codes[:5]:
            st = "Обрабатывается" if num[1] == "taken" else "В очереди"
            text += f"├─ <code>{num[0]}</code> — {st}\n"
    
    if active_qrs:
        text += "\n<b>QR в работе:</b>\n"
        for qr in active_qrs[:5]:
            st = "Обрабатывается" if qr[1] == "taken" else "В очереди"
            text += f"├─ QR #{qr[0]} — {st}\n"
    
    if lt: text += f"\n\n<b>Последняя операция</b>\n└─ {lt[1]}: ${lt[0]:.2f}"
    if pw > 0: text += f"\n\nЗаявок на вывод: {pw}"
    
    markup = types.InlineKeyboardMarkup(row_width=1)
    if not is_vip(message.chat.id):
        vip_price = float(get_setting('vip_price', '5.00'))

    markup.add(types.InlineKeyboardButton("◀️ В меню", callback_data='back_to_menu'))

        bot.edit_message_text(text, message.chat.id, message.message_id, parse_mode='HTML', reply_markup=markup)
    except Exception as e:
        logger.error(f"Ошибка отображения профиля: {e}")
        try:
            bot.send_message(message.chat.id, text, parse_mode='HTML', reply_markup=markup)
        except:
            bot.send_message(message.chat.id, "❌ Ошибка загрузки профиля.", reply_markup=markup)

            types.InlineKeyboardButton("🔐 Пароль", callback_data=f'code_password_{oid}'),
            types.InlineKeyboardButton("⏭ Пропустить", callback_data=f'skip_code_{oid}')
        )

def handle_code_success(call):
    oid = int(call.data.split('_')[2])
    thread_id = call.message.message_thread_id if hasattr(call.message, 'message_thread_id') else None
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    c.execute('SELECT user_id, phone_number FROM code_queue WHERE id=?', (oid,))
    order = c.fetchone()
    if order:
        price = float(get_setting('code_price','4.00'))
        ht = int(get_setting('hold_time','300'))
        c.execute('UPDATE code_queue SET status="completed", completed_at=CURRENT_TIMESTAMP WHERE id=?', (oid,))
        c.execute('UPDATE users SET total_orders=total_orders+1, hold_balance=hold_balance+? WHERE user_id=?', (price, order[0]))
        conn.commit()
        conn.close()
        try: bot.send_message(order[0],
            f"🎉 <b>НОМЕР УСПЕШНО ВСТАЛ!</b>\n\n📱 <code>{order[1]}</code>\n✅ Статус: Подтверждено\n\n💰 Сумма в холде: <b>${price:.2f}</b>\n⏱ Холд: {ht//60} мин\n\n💎 Средства поступят через {ht//60} минут",
            parse_mode='HTML')
        except: pass
        timer = threading.Timer(ht, process_hold_payment, args=[order[0], oid, price, 'code'])
        timer.start()
        if not hasattr(bot, 'hold_timers'): bot.hold_timers = {}
        bot.hold_timers[f'code_{oid}'] = {'timer': timer, 'user_id': order[0], 'amount': price}
        markup = types.InlineKeyboardMarkup()
        markup.add(types.InlineKeyboardButton("❌ Слетел", callback_data=f'code_failed_{oid}'))
        notification_text = f"✅ <b>ПОДТВЕРЖДЕНО</b>\n\n📱 Номер <code>{order[1]}</code>\nуспешно встал"
        if thread_id: send_to_group(notification_text, markup=markup, thread_id=thread_id)
        else:
            try: bot.edit_message_text(notification_text, call.message.chat.id, call.message.message_id, parse_mode='HTML', reply_markup=markup)
            except: send_to_group(notification_text, markup=markup)
        bot.answer_callback_query(call.id, "✅ Подтверждено")
    else: bot.answer_callback_query(call.id, "❌ Заявка не найдена", show_alert=True)

def handle_code_failed(call):
    oid = int(call.data.split('_')[2])
    if hasattr(bot, 'hold_timers') and f'code_{oid}' in bot.hold_timers:
        bot.hold_timers[f'code_{oid}']['timer'].cancel()
        hold_amount = bot.hold_timers[f'code_{oid}']['amount']
        user_id = bot.hold_timers[f'code_{oid}']['user_id']
        conn = sqlite3.connect('max_bot.db')
        c = conn.cursor()
        c.execute('UPDATE users SET hold_balance=hold_balance-? WHERE user_id=?', (hold_amount, user_id))
        conn.commit()
        conn.close()
        del bot.hold_timers[f'code_{oid}']
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    c.execute('SELECT user_id, phone_number FROM code_queue WHERE id=?', (oid,))
    order = c.fetchone()
    if order:
        c.execute('UPDATE code_queue SET status="failed", failed_at=CURRENT_TIMESTAMP WHERE id=?', (oid,))
        conn.commit()
        conn.close()
        try: bot.send_message(order[0], f"❌ <b>НОМЕР СЛЕТЕЛ</b>\n\n📱 <code>{order[1]}</code>\n\n⚠️ Номер не удержался\n💰 Начисление отменено\n\n🔄 Попробуйте другой номер", parse_mode='HTML')
        except: pass
        bot.edit_message_text(f"❌ <b>СЛЕТЕЛ</b>\n\n📱 Номер <code>{order[1]}</code> слетел", call.message.chat.id, call.message.message_id, parse_mode='HTML')

def process_hold_payment(user_id, order_id, amount, order_type='code'):
    try:
        logger.info(f"🕐 Холд завершен! Начисление ${amount} пользователю {user_id} за номер #{order_id} ({order_type})")
        conn = sqlite3.connect('max_bot.db')
        c = conn.cursor()
        if order_type == 'code': c.execute('SELECT status, phone_number FROM code_queue WHERE id=?', (order_id,))
        else: c.execute('SELECT status, completed FROM qr_queue WHERE id=?', (order_id,))
        order = c.fetchone()
        if not order: logger.error(f"❌ Номер #{order_id} не найден"); conn.close(); return
        status = order[0]
        is_completed = False
        if order_type == 'code': is_completed = (status == 'completed')
        else:
            if len(order) > 1: is_completed = (status == 'completed' and order[1] == 1)
            else: is_completed = (status == 'completed')
        logger.info(f"📊 Номер #{order_id}: status={status}, is_completed={is_completed}")
        if is_completed:
            c.execute('UPDATE users SET balance = balance + ?, hold_balance = hold_balance - ?, total_earned = total_earned + ? WHERE user_id = ?', 
                      (amount, amount, amount, user_id))
            if order_type == 'code' and len(order) > 1 and order[1]: desc = f'📱 {order[1]} — ${amount:.2f}'
            else: desc = f'📷 QR-код — ${amount:.2f}'
            c.execute('INSERT INTO transactions (user_id, amount, type, description) VALUES (?,?,"income",?)', (user_id, amount, desc))
            c.execute('UPDATE users SET success_orders=success_orders+1 WHERE user_id=?', (user_id,))
            conn.commit()
            c.execute('SELECT balance, hold_balance FROM users WHERE user_id=?', (user_id,))
            user_data = c.fetchone()
            new_balance = user_data[0]
            hold_balance = user_data[1]
            logger.info(f"✅ Начислено! Баланс: ${new_balance:.2f}, Холд: ${hold_balance:.2f}")
            if order_type == 'code' and len(order) > 1 and order[1]: text = f"💰 <b>СРЕДСТВА ЗАЧИСЛЕНЫ</b>\n\n📱 Номер: <code>{order[1]}</code>\n💵 Сумма: <b>${amount:.2f}</b>\n💎 Баланс: <b>${new_balance:.2f}</b>\n\n✅ Холд завершен"
            else: text = f"💰 <b>СРЕДСТВА ЗАЧИСЛЕНЫ</b>\n\n📷 QR-код\n💵 Сумма: <b>${amount:.2f}</b>\n💎 Баланс: <b>${new_balance:.2f}</b>\n\n✅ Холд завершен"
            try: bot.send_message(user_id, text, parse_mode='HTML')
            except: pass
            pay_referral_bonus(user_id, order_id)
            update_daily_stats(user_id, amount)
            update_weekly_stats(user_id, amount)
        conn.close()
    except Exception as e: logger.error(f"Ошибка начисления: {e}")

@bot.message_handler(func=lambda m: user_states.get(m.from_user.id,{}).get('state')=='waiting_withdraw_amount')
def process_withdraw_amount(message):
    try: amount = float(message.text.strip())
    except: bot.reply_to(message, "❌ Введите число"); return
    del user_states[message.from_user.id]
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    c.execute('SELECT balance, hold_balance FROM users WHERE user_id=?', (message.from_user.id,))
    user = c.fetchone()
    if not user: bot.send_message(message.chat.id, "❌ Ошибка"); conn.close(); return
    avail = user[0] - user[1]
    if amount < 1.0 or amount > avail: bot.send_message(message.chat.id, f"❌ Неверная сумма\n💰 Доступно: ${avail:.2f}"); conn.close(); return
    c.execute('INSERT INTO withdraw_requests (user_id, amount, status) VALUES (?,?,"pending")', (message.from_user.id, amount))
    c.execute('UPDATE users SET balance=balance-?, hold_balance=hold_balance+? WHERE user_id=?', (amount, amount, message.from_user.id))
    wid = c.lastrowid
    conn.commit()
    conn.close()
    bot.send_message(message.chat.id, f"✅ <b>ЗАЯВКА СОЗДАНА</b>\n\n📋 Номер: #{wid}\n💳 Сумма: ${amount:.2f}\n📅 {get_msk_time().strftime('%d.%m.%Y %H:%M')} МСК\n\n⏱ Статус: Ожидает обработки\n👑 Администратор проверит заявку", parse_mode='HTML')
    for aid in ADMIN_IDS:
        try:
            markup = types.InlineKeyboardMarkup()
            markup.add(types.InlineKeyboardButton("✅ Выплатить", callback_data=f'process_withdraw_{wid}'))
            bot.send_message(aid, f"💰 <b>НОВАЯ ВЫПЛАТА</b>\n\n📋 #{wid}\n👤 @{message.from_user.username or 'Нет'}\n🆔 <code>{message.from_user.id}</code>\n💳 ${amount:.2f}", parse_mode='HTML', reply_markup=markup)
        except: pass

@bot.message_handler(func=lambda m: user_states.get(m.from_user.id,{}).get('state')=='admin_editing_balance')
def process_admin_edit_balance(message):
    if not is_admin(message.from_user.id): return
    us = user_states[message.from_user.id]
    uid = us['user_id']
    del user_states[message.from_user.id]
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    c.execute('SELECT balance, username FROM users WHERE user_id=?', (uid,))
    user = c.fetchone()
    if not user: bot.reply_to(message, "❌ Не найден"); conn.close(); return
    old_balance = user[0]
    inp = message.text.strip()
    try:
        if inp.startswith('+'): new_balance = old_balance + float(inp[1:]); diff = float(inp[1:]); action = "пополнен"
        elif inp.startswith('-'): diff = float(inp[1:]); new_balance = max(0, old_balance - diff); action = "уменьшен"
        else: new_balance = float(inp); diff = new_balance - old_balance; action = "установлен"
    except: bot.reply_to(message, "❌ Неверный формат"); conn.close(); return
    c.execute('UPDATE users SET balance=? WHERE user_id=?', (new_balance, uid))
    if diff != 0: c.execute('INSERT INTO transactions (user_id, amount, type, description) VALUES (?,?,"admin_edit",?)', (uid, abs(diff), f'Админ {action} баланс на ${abs(diff):.2f}'))
    conn.commit()
    conn.close()
    try: bot.send_message(uid, f"💰 <b>Баланс изменен</b>\n\n📝 Старый: ${old_balance:.2f}\n💎 Новый: <b>${new_balance:.2f}</b>\n📊 Изменение: {'+' if diff > 0 else ''}${diff:.2f}", parse_mode='HTML')
    except: pass
    bot.reply_to(message, f"✅ Баланс обновлен!\n👤 @{user[1] or 'Нет'}\n💰 ${old_balance:.2f} → <b>${new_balance:.2f}</b>", parse_mode='HTML', reply_markup=types.InlineKeyboardMarkup().add(types.InlineKeyboardButton("🔍 Поиск", callback_data='admin_search'), types.InlineKeyboardButton("◀️ В админ-панель", callback_data='admin_back')))

        c.execute('UPDATE qr_queue SET status="problem", problem_at=CURRENT_TIMESTAMP WHERE id=?', (qr_id,))
        conn.commit()
        conn.close()
        try: bot.send_message(qr[0], "⚠️ <b>ПРОБЛЕМА С QR</b>\n\n❌ QR недействителен\n🔄 Попробуйте другой", parse_mode='HTML')
        except: pass
        bot.edit_message_text(f"⏭ QR пропущен\n📋 #{qr_id}", call.message.chat.id, call.message.message_id, parse_mode='HTML')
        bot.answer_callback_query(call.id, "⏭ Пропущен")
    else: bot.answer_callback_query(call.id, "❌ Не найдена", show_alert=True)

@bot.callback_query_handler(func=lambda call: True)
def main_callback(call):
    data = call.data
    if data == 'sell_max': show_sell_options(call.message)
    elif data == 'sell_code': start_code_sell(call.message)
    elif data == 'sell_qr': start_qr_sell(call.message)
    elif data == 'confirm_qr': confirm_qr_sell(call)
    elif data == 'profile': show_profile(call.message)
    elif data == 'back_to_menu': show_main_menu(call.message.chat.id, call.message.message_id)
    elif data == 'back_to_sell': show_sell_options(call.message)
    elif data == 'top_daily': show_top_daily(call)
    elif data == 'top_weekly': show_top_weekly(call)
    elif data == 'get_code': get_next_code(call)
    elif data == 'get_qr': get_next_qr(call)
    elif data.startswith('send_code_'): handle_send_code(call)
    elif data.startswith('skip_code_'): handle_skip_code(call)
    elif data.startswith('enter_code_'): handle_enter_code_request(call)
    elif data.startswith('code_success_'): handle_code_success(call)
    elif data.startswith('code_password_'): handle_code_password(call)
    elif data.startswith('code_failed_'): handle_code_failed(call)
    elif data.startswith('scanned_'): handle_qr_scanned(call)
    elif data.startswith('qr_success_'): handle_qr_success(call)
    elif data.startswith('qr_password_'): handle_qr_password(call)
    elif data.startswith('qr_failed_'): handle_qr_failed(call)
    elif data.startswith('resend_code_'): handle_resend_code(call)
    elif data.startswith('resend_qr_'): handle_resend_qr(call)
    elif data.startswith('send_qr_'): handle_send_qr_again(call)
    elif data.startswith('cancel_qr_'): handle_cancel_qr(call)
    elif data == 'waiting': bot.answer_callback_query(call.id, "⏳ Ожидайте...")
