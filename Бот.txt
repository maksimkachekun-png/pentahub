def is_admin(uid): return uid in ADMIN_IDS
def get_setting(key, default=None):
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    c.execute('SELECT value FROM settings WHERE key=?', (key,))
    r = c.fetchone()
    conn.close()
    return r[0] if r else default

def update_setting(key, value):
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    c.execute('INSERT OR REPLACE INTO settings (key,value,updated_at) VALUES (?,?,CURRENT_TIMESTAMP)', (key, str(value)))
    conn.commit()
    conn.close()

def check_subscription(uid):
    try:
        member = bot.get_chat_member(CHANNEL_ID, uid)
        return member.status in ['member','administrator','creator']
    except Exception as e:
        logger.error(f"Ошибка проверки подписки: {e}")
        return False

def get_queue_position(uid, qtype='code'):
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    if qtype == 'code':
        c.execute('SELECT COUNT(*) FROM code_queue WHERE status="waiting" AND id<=(SELECT COALESCE(MIN(id),0) FROM code_queue WHERE user_id=? AND status="waiting")', (uid,))
    else:
        c.execute('SELECT COUNT(*) FROM qr_queue WHERE status="waiting" AND id<=(SELECT COALESCE(MIN(id),0) FROM qr_queue WHERE user_id=? AND status="waiting")', (uid,))
    p = c.fetchone()[0]
    conn.close()
    return p

def add_to_queue(uid, qtype='code', phone=None):
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    if qtype == 'code':
        c.execute('SELECT id FROM code_queue WHERE user_id=? AND status="waiting"', (uid,))
        if c.fetchone():
            conn.close()
            return False, "Вы уже в очереди"
        c.execute('INSERT INTO code_queue (user_id,phone_number,status) VALUES (?,?,"waiting")', (uid, phone))
    else:
        c.execute('SELECT id FROM qr_queue WHERE user_id=? AND status="waiting"', (uid,))
        if c.fetchone():
            conn.close()
            return False, "Вы уже в очереди QR"
        c.execute('INSERT INTO qr_queue (user_id,status) VALUES (?,"waiting")', (uid,))
    conn.commit()
    conn.close()
    return True, "Добавлены в очередь"

def show_profile(message):

    user = c.fetchone()
    if not user: show_main_menu(message.chat.id, message.message_id); conn.close(); return
    
    c.execute('''
        SELECT 
            COUNT(*) as total,
            SUM(CASE WHEN status="completed" THEN 1 ELSE 0 END) as completed,
            SUM(CASE WHEN status="failed" THEN 1 ELSE 0 END) as failed,
            SUM(CASE WHEN status IN ("taken","waiting") THEN 1 ELSE 0 END) as in_progress
        FROM code_queue WHERE user_id=?
    ''', (message.chat.id,))
    code_stats = c.fetchone()
    
    c.execute('''
        SELECT 
            COUNT(*) as total,
            SUM(CASE WHEN status="completed" AND completed=1 THEN 1 ELSE 0 END) as completed,
            SUM(CASE WHEN status="failed" THEN 1 ELSE 0 END) as failed,
            SUM(CASE WHEN status IN ("taken","waiting") THEN 1 ELSE 0 END) as in_progress

def handle_code_success(call):
    oid = int(call.data.split('_')[2])

    conn = sqlite3.connect('max_bot.db')

    order = c.fetchone()
    if order:
        price = float(get_setting('code_price','4.00'))
        ht = int(get_setting('hold_time','300'))
        c.execute('UPDATE code_queue SET status="completed", completed_at=CURRENT_TIMESTAMP WHERE id=?', (oid,))
        c.execute('UPDATE users SET total_orders=total_orders+1 WHERE user_id=?', (order[0],))
        conn.commit()
        conn.close()
        try: bot.send_message(order[0],
            f"🎉 **НОМЕР УСПЕШНО ВСТАЛ!**\n\n"
            f"📱 `{order[1]}`\n"
            f"✅ Статус: Подтверждено\n\n"

            f"💎 Средства поступят через {ht//60} минут",
            parse_mode='Markdown')
        except: pass
        timer = threading.Timer(ht, process_hold_payment, args=[order[0], oid, price, 'code'])
        timer.start()
        if not hasattr(bot, 'hold_timers'): bot.hold_timers = {}
        bot.hold_timers[f'code_{oid}'] = {'timer': timer, 'user_id': order[0], 'amount': price}
        markup = types.InlineKeyboardMarkup()
        markup.add(types.InlineKeyboardButton("❌ Слетел", callback_data=f'code_failed_{oid}'))

            f"✅ **ПОДТВЕРЖДЕНО**\n\n"
            f"📱 Номер `{order[1]}`\n"
            f"успешно встал"
        )
        if thread_id: send_to_group(notification_text, markup=markup, thread_id=thread_id)
        else:
            try: bot.edit_message_text(notification_text, call.message.chat.id, call.message.message_id, parse_mode='Markdown', reply_markup=markup)
            except: send_to_group(notification_text, markup=markup)
        bot.answer_callback_query(call.id, "✅ Подтверждено")
    else: bot.answer_callback_query(call.id, "❌ Заявка не найдена", show_alert=True)

def process_hold_payment(user_id, order_id, amount, order_type='code'):
    try:
        logger.info(f"🕐 Холд завершен! Начисление ${amount} пользователю {user_id} за заказ #{order_id} ({order_type})")
        conn = sqlite3.connect('max_bot.db')
        c = conn.cursor()
        if order_type == 'code': c.execute('SELECT status, phone_number FROM code_queue WHERE id=?', (order_id,))
        else: c.execute('SELECT status, completed FROM qr_queue WHERE id=?', (order_id,))
        order = c.fetchone()
        if not order: logger.error(f"❌ Заявка #{order_id} не найдена"); conn.close(); return
        status = order[0]
        is_completed = False
        if order_type == 'code': is_completed = (status == 'completed')
        else:
            if len(order) > 1: is_completed = (status == 'completed' and order[1] == 1)
            else: is_completed = (status == 'completed')
        if is_completed:
            c.execute('UPDATE users SET balance=balance+?, total_earned=total_earned+? WHERE user_id=?', (amount, amount, user_id))
            if order_type == 'code' and len(order) > 1 and order[1]: desc = f'📱 {order[1]} — ${amount:.2f}'
            else: desc = f'📷 QR-код — ${amount:.2f}'
            c.execute('INSERT INTO transactions (user_id, amount, type, description) VALUES (?,?,"income",?)', (user_id, amount, desc))
            c.execute('UPDATE users SET success_orders=success_orders+1 WHERE user_id=?', (user_id,))
            conn.commit()
            c.execute('SELECT balance, hold_balance FROM users WHERE user_id=?', (user_id,))
            user_data = c.fetchone()
            new_balance = user_data[0]
            if order_type == 'code' and len(order) > 1 and order[1]: text = f"💰 **СРЕДСТВА ЗАЧИСЛЕНЫ**\n\n📱 Номер: `{order[1]}`\n💵 Сумма: **${amount:.2f}**\n💎 Баланс: **${new_balance:.2f}**\n\n✅ Холд завершен"
            else: text = f"💰 **СРЕДСТВА ЗАЧИСЛЕНЫ**\n\n📷 QR-код\n💵 Сумма: **${amount:.2f}**\n💎 Баланс: **${new_balance:.2f}**\n\n✅ Холд завершен"
            try: bot.send_message(user_id, text, parse_mode='Markdown')
            except: pass
        conn.close()
    except Exception as e: logger.error(f"Ошибка начисления: {e}")

@bot.message_handler(func=lambda m: user_states.get(m.from_user.id,{}).get('state')=='admin_searching')
def admin_search_exec(message):
    if not is_admin(message.from_user.id): return
    q = message.text.strip()
    del user_states[message.from_user.id]
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    if q.startswith('@'): c.execute('SELECT * FROM users WHERE username=?', (q[1:],))
    else:
        try: c.execute('SELECT * FROM users WHERE user_id=?', (int(q),))
        except: bot.reply_to(message, "❌ Неверный формат"); conn.close(); return
    user = c.fetchone()
    if user:
        c.execute('SELECT COUNT(*) as t, SUM(CASE WHEN status="completed" THEN 1 ELSE 0 END) as ok, SUM(CASE WHEN status="failed" THEN 1 ELSE 0 END) as fl, SUM(CASE WHEN status="problem" THEN 1 ELSE 0 END) as pr, SUM(CASE WHEN status IN ("taken","waiting") THEN 1 ELSE 0 END) as ac FROM code_queue WHERE user_id=?', (user[0],))
        cs = c.fetchone()
        c.execute('SELECT COUNT(*) as t, SUM(CASE WHEN status="completed" THEN 1 ELSE 0 END) as ok, SUM(CASE WHEN status="failed" THEN 1 ELSE 0 END) as fl, SUM(CASE WHEN status="problem" THEN 1 ELSE 0 END) as pr, SUM(CASE WHEN status IN ("taken","waiting") THEN 1 ELSE 0 END) as ac FROM qr_queue WHERE user_id=?', (user[0],))
        qs = c.fetchone()
        c.execute('SELECT amount, type, description, created_at FROM transactions WHERE user_id=? ORDER BY created_at DESC LIMIT 5', (user[0],))
        transactions = c.fetchall()
        c.execute('SELECT COUNT(*), SUM(amount) FROM withdraw_requests WHERE user_id=? AND status="pending"', (user[0],))
        pending_w = c.fetchone()
        conn.close()
        total_orders = (cs[0] or 0) + (qs[0] or 0)
        success = (cs[1] or 0) + (qs[1] or 0)
        failed = (cs[2] or 0) + (qs[2] or 0)
        problem = (cs[3] or 0) + (qs[3] or 0)
        active = (cs[4] or 0) + (qs[4] or 0)
        success_rate = (success/total_orders*100) if total_orders > 0 else 0
        text = (
            f"✅ **ПОЛЬЗОВАТЕЛЬ НАЙДЕН**\n\n"
            f"📱 **Информация**\n"
            f"• ID: `{user[0]}`\n"
            f"• Username: @{user[1] or 'Не указан'}\n"
            f"• Статус: {'🚫 Заблокирован' if user[11] else '✅ Активен'}\n"
            f"• Регистрация: {user[9][:19] if user[9] else 'Н/Д'}\n\n"
            f"💰 **Финансы**\n"
            f"• Баланс: **${user[2]:.2f}**\n"
            f"• В холде: ${user[4]:.2f}\n"
            f"• Заработано: ${user[3]:.2f}\n"
            f"• Выведено: ${user[5]:.2f}\n\n"
            f"📊 **Заказы**\n"
            f"• Всего: {total_orders}\n"
            f"• Успешно: {success} ({success_rate:.0f}%)\n"
            f"• Провалено: {failed}\n"
            f"• Проблемных: {problem}\n"
            f"• Активных: {active}\n"
        )
        if pending_w[0] > 0: text += f"\n💳 **Выводы**\n• Активных: {pending_w[0]}\n• На сумму: ${pending_w[1] or 0:.2f}\n"
        if transactions:
            text += f"\n💼 **Последние транзакции:**\n"
            for t in transactions[:3]:
                emoji = "🟢" if t[1] == "income" else "🔴" if t[1] == "withdraw" else "⚪"
                text += f"{emoji} ${t[0]:.2f} — {t[2]}\n"
        markup = types.InlineKeyboardMarkup(row_width=2)
        markup.add(types.InlineKeyboardButton("💰 Изменить баланс", callback_data=f'admin_edit_balance_{user[0]}'), types.InlineKeyboardButton("📱 Заказы", callback_data=f'admin_user_orders_{user[0]}'))
        if user[11]: markup.add(types.InlineKeyboardButton("✅ Разблокировать", callback_data=f'admin_unblock_{user[0]}'))
        else: markup.add(types.InlineKeyboardButton("🚫 Заблокировать", callback_data=f'admin_block_{user[0]}'))
        markup.add(types.InlineKeyboardButton("📩 Отправить сообщение", callback_data=f'admin_msg_{user[0]}'))
        markup.add(types.InlineKeyboardButton("🔍 Новый поиск", callback_data='admin_search'))
        markup.add(types.InlineKeyboardButton("◀️ В админ-панель", callback_data='admin_back'))
        bot.reply_to(message, text, parse_mode='Markdown', reply_markup=markup)
    else:
        conn.close()
        markup = types.InlineKeyboardMarkup()
        markup.add(types.InlineKeyboardButton("🔍 Попробовать снова", callback_data='admin_search'), types.InlineKeyboardButton("◀️ В админ-панель", callback_data='admin_back'))
        bot.reply_to(message, "❌ **Пользователь не найден**\n\nПроверьте правильность ID или @username", parse_mode='Markdown', reply_markup=markup)

@bot.message_handler(func=lambda m: user_states.get(m.from_user.id,{}).get('state')=='admin_editing_balance')
def process_admin_edit_balance(message):
    if not is_admin(message.from_user.id): return
    us = user_states[message.from_user.id]
    uid = us['user_id']
    del user_states[message.from_user.id]
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    c.execute('SELECT balance, username FROM users WHERE user_id=?', (uid,))
    user = c.fetchone()
    if not user: bot.reply_to(message, "❌ Не найден"); conn.close(); return
    old_balance = user[0]
    inp = message.text.strip()
    try:
        if inp.startswith('+'): new_balance = old_balance + float(inp[1:]); diff = float(inp[1:]); action = "пополнен"
        elif inp.startswith('-'): diff = float(inp[1:]); new_balance = max(0, old_balance - diff); action = "уменьшен"
        else: new_balance = float(inp); diff = new_balance - old_balance; action = "установлен"
    except: bot.reply_to(message, "❌ Неверный формат"); conn.close(); return
    c.execute('UPDATE users SET balance=? WHERE user_id=?', (new_balance, uid))
    if diff != 0: c.execute('INSERT INTO transactions (user_id, amount, type, description) VALUES (?,?,"admin_edit",?)', (uid, abs(diff), f'Админ {action} баланс на ${abs(diff):.2f}'))
    conn.commit()
    conn.close()
    try: bot.send_message(uid, f"💰 **Баланс изменен**\n\n📝 Старый: ${old_balance:.2f}\n💎 Новый: **${new_balance:.2f}**\n📊 Изменение: {'+' if diff > 0 else ''}${diff:.2f}", parse_mode='Markdown')
    except: pass
    bot.reply_to(message, f"✅ Баланс обновлен!\n👤 @{user[1] or 'Нет'}\n💰 ${old_balance:.2f} → **${new_balance:.2f}**", parse_mode='Markdown', reply_markup=types.InlineKeyboardMarkup().add(types.InlineKeyboardButton("🔍 Поиск", callback_data='admin_search'), types.InlineKeyboardButton("◀️ В админ-панель", callback_data='admin_back')))

        conn.close()
        if qr[1]:
            try: bot.send_message(qr[1], f"❌ QR отменен\n📋 #{qr_id}\nВозвращен в очередь", parse_mode='Markdown')
            except: pass
        try: bot.edit_message_caption("❌ QR отменен\n\nЗаявка возвращена в очередь", call.message.chat.id, call.message.message_id, parse_mode='Markdown', reply_markup=types.InlineKeyboardMarkup().add(types.InlineKeyboardButton("◀️ В меню", callback_data='back_to_menu')))
        except: bot.send_message(call.message.chat.id, "❌ QR отменен", parse_mode='Markdown')
        bot.answer_callback_query(call.id, "✅ Отменен")
    else: bot.answer_callback_query(call.id, "❌ Не найдена", show_alert=True)

@bot.callback_query_handler(func=lambda call: call.data.startswith('resend_code_'))
def handle_resend_code(call):
    oid = int(call.data.split('_')[2])
    conn = sqlite3.connect('max_bot.db')
    c = conn.cursor()
    c.execute('SELECT id, taken_by, phone_number FROM code_queue WHERE id=? AND status="taken"', (oid,))
    order = c.fetchone()
    if order and order[1]:
        c.execute('UPDATE code_queue SET attempts=attempts+1 WHERE id=?', (oid,))
        conn.commit()
        try: bot.send_message(order[1], f"🔄 **ПОВТОРНЫЙ КОД**\n\n📱 `{order[2]}`\n📋 #{oid}\n\nОтправьте код повторно", parse_mode='Markdown')
        except: pass
        bot.answer_callback_query(call.id, "✅ Запрос отправлен!")
        markup = types.InlineKeyboardMarkup(row_width=1)
        markup.add(types.InlineKeyboardButton("🔢 Ввести код", callback_data=f'enter_code_{oid}'), types.InlineKeyboardButton("🔄 Повтор", callback_data=f'resend_code_{oid}'))
        bot.edit_message_text("🔄 Запрос на повторный код отправлен\n⏱ Ожидайте...", call.message.chat.id, call.message.message_id, parse_mode='Markdown', reply_markup=markup)
    else: bot.answer_callback_query(call.id, "❌ Не активна", show_alert=True)
    conn.close()

@bot.callback_query_handler(func=lambda call: True)
def main_callback(call):
    data = call.data
    if data == 'sell_max': show_sell_options(call.message)
    elif data == 'sell_code': start_code_sell(call.message)
    elif data == 'sell_qr': start_qr_sell(call.message)
    elif data == 'confirm_qr': confirm_qr_sell(call)
    elif data == 'profile': show_profile(call.message)
    elif data == 'back_to_menu': show_main_menu(call.message.chat.id, call.message.message_id)
    elif data == 'back_to_sell': show_sell_options(call.message)
    elif data == 'get_code': get_next_code(call)
    elif data == 'get_qr': get_next_qr(call)
    elif data.startswith('send_code_'): handle_send_code(call)
    elif data.startswith('skip_code_'): handle_skip_code(call)
    elif data.startswith('enter_code_'): handle_enter_code_request(call)
    elif data.startswith('code_success_'): handle_code_success(call)
    elif data.startswith('code_password_'): handle_code_password(call)
    elif data.startswith('code_failed_'): handle_code_failed(call)
    elif data.startswith('scanned_'): handle_qr_scanned(call)
    elif data.startswith('qr_success_'): handle_qr_success(call)
    elif data.startswith('qr_password_'): handle_qr_password(call)
    elif data.startswith('qr_failed_'): handle_qr_failed(call)
    elif data == 'waiting': bot.answer_callback_query(call.id, "⏳ Ожидайте...")
